此文档为中国地图部分的渲染实现过程
# 关键技术要点
1. ECharts地图注册：
- 需要先加载中国地图的JSON文件
- 通过 echarts.registerMap('china', chinaJson)注册
2. 数据映射策略：
- 将数据库中的省份名称映射到ECharts可识别的名称
- 处理特殊行政区划（如内蒙古自治区→内蒙古）
3. 性能优化：
- 使用Web Worker处理大量数据计算
4. 交互设计：
- 点击下钻 + 返回功能
- 鼠标悬停高亮


| 步骤 | 用户操作 | 前端响应 | 后端处理 |
|------|----------|----------|----------|
| 1 | 页面加载 | 初始化ECharts实例 | 请求 `/api/map-data/` |
| 2 | 选择年份 | 重新请求API | 查询该年份各省数据 |
| 3 | 点击地图省份 | 触发click事件 | 请求 `/api/province-detail/` |
| 4 | 选择指标 | 重新渲染地图 | 重新计算比值 |
| 5 | 悬停省份 | 显示Tooltip | 实时计算显示值 |

// static/js/dashboard.js - 地图初始化与渲染函数
function initChinaMap() {
    // 1. 初始化ECharts实例
    const mapDom = document.getElementById('china-map');
    const mapChart = echarts.init(mapDom, null, {
        renderer: 'canvas',
        width: 'auto',
        height: 500
    });
    
    // 2. 定义地图配置选项
    const mapOption = {
        // 标题
        title: {
            text: '全国城乡收入差异分布',
            subtext: '颜色越深表示城乡收入差距越大',
            left: 'center',
            textStyle: { fontSize: 16 }
        },
        
        // 提示框配置
        tooltip: {
            trigger: 'item',
            backgroundColor: 'rgba(255, 255, 255, 0.95)',
            borderColor: '#ddd',
            borderWidth: 1,
            padding: [10, 15],
            textStyle: { color: '#333' },
            formatter: function(params) {
                // 自定义提示框内容
                if (params.value) {
                    return `
                    <div style="font-weight: bold; margin-bottom: 5px;">
                        ${params.name}
                    </div>
                    <div style="color: #666;">
                        城乡收入比: <b style="color: #1890ff;">${params.value}</b><br/>
                        （城镇收入 ÷ 农村收入）
                    </div>
                    `;
                } else {
                    return `${params.name}<br/>暂无数据`;
                }
            }
        },
        
        // 视觉映射组件（右侧颜色条）
        visualMap: {
            type: 'continuous',
            min: 1.5,    // 最小比值
            max: 3.5,    // 最大比值
            left: 20,
            bottom: 20,
            text: ['差距大', '差距小'],
            textStyle: { color: '#666' },
            calculable: true,
            inRange: {
                // 渐变色系：蓝 -> 青 -> 黄 -> 红
                color: ['#e6f7ff', '#91d5ff', '#1890ff', '#096dd9', '#0050b3']
            },
            // 分段显示
            pieces: [
                { min: 1.5, max: 2.0, label: '差距较小' },
                { min: 2.0, max: 2.5, label: '差距中等' },
                { min: 2.5, max: 3.0, label: '差距较大' },
                { min: 3.0, max: 3.5, label: '差距很大' }
            ]
        },
        
        // 地理坐标系配置
        geo: {
            map: 'china',
            roam: true,  // 允许缩放和平移
            zoom: 1.2,   // 初始缩放级别
            center: [105, 38],  // 地图中心点 [经度, 纬度]
            label: {
                show: true,
                fontSize: 10,
                color: 'rgba(0, 0, 0, 0.7)'
            },
            itemStyle: {
                areaColor: '#f5f5f5',  // 默认区域颜色
                borderColor: '#ccc',
                borderWidth: 0.5
            },
            emphasis: {  // 高亮状态样式
                itemStyle: {
                    areaColor: '#ffd700',  // 金色高亮
                    borderColor: '#333',
                    borderWidth: 1
                },
                label: {
                    fontWeight: 'bold',
                    color: '#000'
                }
            }
        },
        
        // 数据系列
        series: [
            {
                name: '城乡收入比',
                type: 'map',
                map: 'china',
                coordinateSystem: 'geo',
                data: [],  // 初始为空，通过API加载
                
                // 数据项样式
                itemStyle: {
                    borderColor: '#fff',
                    borderWidth: 1
                },
                
                // 标签显示
                label: {
                    show: true,
                    position: 'inside',  // 标签在区域内
                    formatter: function(params) {
                        // 只显示简称
                        const shortName = getProvinceShortName(params.name);
                        return shortName;
                    },
                    fontSize: 10,
                    color: '#fff',
                    textBorderColor: '#000',
                    textBorderWidth: 1
                },
                
                // 高亮状态
                emphasis: {
                    itemStyle: {
                        areaColor: '#f0ad4e',  // 悬停颜色
                        shadowBlur: 10,
                        shadowColor: 'rgba(0, 0, 0, 0.5)'
                    }
                },
                
                // 选中状态
                select: {
                    itemStyle: {
                        areaColor: '#d9534f'  // 选中颜色
                    }
                }
            }
        ]
    };
    
    // 3. 加载地图数据
    $.ajax({
        url: '/api/map-data/',
        type: 'GET',
        data: { 
            year: $('#year-select').val(),
            indicator: 'income_ratio' 
        },
        success: function(response) {
            // 4. 处理数据格式转换
            const mapData = response.data.map(item => {
                return {
                    name: item.province,
                    value: item.ratio
                };
            });
            
            // 5. 更新图表数据
            mapOption.series[0].data = mapData;
            mapChart.setOption(mapOption);
            
            // 6. 触发地图渲染完成事件
            $(document).trigger('map:rendered');
        },
        error: function() {
            console.error('地图数据加载失败');
        }
    });
    
    // 7. 绑定地图点击事件（省份下钻）
    mapChart.on('click', function(params) {
        if (params.componentType === 'series') {
            const provinceName = params.name;
            
            // 显示加载状态
            showLoading(`正在加载 ${provinceName} 数据...`);
            
            // 发起省份数据请求
            $.ajax({
                url: '/api/province-detail/',
                data: {
                    province: provinceName,
                    year: $('#year-select').val()
                },
                success: function(provinceData) {
                    // 更新页面标题
                    $('#current-province').text(provinceName)
                        .removeClass('bg-info')
                        .addClass('bg-warning')
                        .append(' <i class="fas fa-undo-alt" id="back-to-national"></i>');
                    
                    // 更新所有卡片数据
                    updateAllCards(provinceData);
                    
                    // 添加返回全国按钮事件
                    $('#back-to-national').click(function() {
                        resetToNationalView();
                    });
                },
                complete: function() {
                    hideLoading();
                }
            });
        }
    });
    
    // 8. 窗口大小变化时重绘
    window.addEventListener('resize', function() {
        mapChart.resize();
    });
    
    return mapChart;
}

// 省份名称简写转换
function getProvinceShortName(fullName) {
    const mapping = {
        '北京市': '北京',
        '天津市': '天津',
        '上海市': '上海',
        '重庆市': '重庆',
        '内蒙古自治区': '内蒙古',
        '广西壮族自治区': '广西',
        '西藏自治区': '西藏',
        '宁夏回族自治区': '宁夏',
        '新疆维吾尔自治区': '新疆',
        '香港特别行政区': '香港',
        '澳门特别行政区': '澳门'
    };
    
    // 移除"省"和"市"后缀
    let shortName = mapping[fullName] || fullName;
    shortName = shortName.replace('省', '').replace('市', '');
    
    return shortName;
}

# views.py - 地图数据API
from django.http import JsonResponse
from django.db.models import Q, F
from .models import EconomicData

def map_data_api(request):
    """提供地图所需的数据"""
    # 获取查询参数
    year = request.GET.get('year', 2024)
    indicator_type = request.GET.get('indicator', 'income_ratio')
    
    # 查询该年份所有省份的数据
    provinces_data = EconomicData.objects.filter(
        year=year
    ).exclude(
        province='全国'  # 排除全国数据
    ).values('province').distinct()
    
    result_data = []
    
    for province_info in provinces_data:
        province = province_info['province']
        
        # 根据指标类型计算比值
        if indicator_type == 'income_ratio':
            # 计算城乡收入比
            urban_income = EconomicData.objects.filter(
                year=year,
                province=province,
                indicator__contains='城镇居民人均可支配收入'
            ).first()
            
            rural_income = EconomicData.objects.filter(
                year=year,
                province=province,
                indicator__contains='农村居民人均可支配收入'
            ).first()
            
            if urban_income and rural_income and rural_income.value != 0:
                ratio = round(urban_income.value / rural_income.value, 2)
            else:
                ratio = None
                
        elif indicator_type == 'consumption_ratio':
            # 计算城乡消费比（类似逻辑）
            pass
        
        # 构建返回数据
        if ratio is not None:
            result_data.append({
                'province': province,
                'ratio': ratio,
                'year': year,
                'indicator': indicator_type
            })
    
    return JsonResponse({
        'success': True,
        'year': year,
        'indicator': indicator_type,
        'data': result_data,
        'count': len(result_data)
    })

def province_detail_api(request):
    """省份下钻详情数据"""
    province = request.GET.get('province')
    year = request.GET.get('year', 2024)
    
    if not province:
        return JsonResponse({'error': '省份参数缺失'}, status=400)
    
    # 获取该省份所有指标数据
    province_data = EconomicData.objects.filter(
        province=province,
        year=year
    )
    
    # 提取关键指标
    indicators = {
        'urban_income': None,  # 城镇收入
        'rural_income': None,  # 农村收入
        'urban_expenditure': None,  # 城镇支出
        'rural_expenditure': None,  # 农村支出
        'urban_population': None,  # 城镇人口
        'rural_population': None,  # 农村人口
    }
    
    for item in province_data:
        indicator_name = item.indicator
        
        if '城镇居民人均可支配收入' in indicator_name:
            indicators['urban_income'] = item.value
        elif '农村居民人均可支配收入' in indicator_name:
            indicators['rural_income'] = item.value
        elif '城镇居民人均消费支出' in indicator_name:
            indicators['urban_expenditure'] = item.value
        elif '农村居民人均消费支出' in indicator_name:
            indicators['rural_expenditure'] = item.value
        elif '城镇人口' in indicator_name or '人口普查城镇' in indicator_name:
            indicators['urban_population'] = item.value
        elif '农村人口' in indicator_name or '人口普查乡村' in indicator_name:
            indicators['rural_population'] = item.value
    
    # 计算比率
    income_ratio = None
    if indicators['urban_income'] and indicators['rural_income']:
        income_ratio = round(indicators['urban_income'] / indicators['rural_income'], 2)
    
    expenditure_ratio = None
    if indicators['urban_expenditure'] and indicators['rural_expenditure']:
        expenditure_ratio = round(indicators['urban_expenditure'] / indicators['rural_expenditure'], 2)
    
    return JsonResponse({
        'province': province,
        'year': year,
        'indicators': indicators,
        'ratios': {
            'income_ratio': income_ratio,
            'expenditure_ratio': expenditure_ratio
        },
        'ranking': {
            'income_rank': calculate_province_ranking(province, year, 'income'),
            'expenditure_rank': calculate_province_ranking(province, year, 'expenditure')
        }
    })
